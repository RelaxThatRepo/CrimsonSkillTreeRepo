43 FName ActorResourcePropertyName;
44
45 // The gameplay attribute to use as the resource pool.
46 UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Cost Definition", meta = (EditCondition = "CostSource == ENodeCostSource::GameplayAttributeValue"))
47 FGameplayAttribute ResourceGameplayAttribute;
48
49 // Equality operator for TMap support.
50 bool operator==(const FNodeCostDefinition& Other) const
51 {
52 if (CostSource != Other.CostSource) return false;
53 switch (CostSource)
54 {
55 case ENodeCostSource::ActorIntegerProperty:
56 return ActorResourcePropertyName == Other.ActorResourcePropertyName;
57 case ENodeCostSource::GameplayAttributeValue:
58 return ResourceGameplayAttribute == Other.ResourceGameplayAttribute;
59 case ENodeCostSource::NoCost:
60 default:
61 return true;
62 }
63 }
64
65 // Helper function to get a clean resource name, with fallbacks.
66 FText GetResourceName() const
67 {
68 if (!ResourceUserFacingName.IsEmpty())
69 {
70 return ResourceUserFacingName;
71 }
72 if (CostSource == ENodeCostSource::ActorIntegerProperty && ActorResourcePropertyName != NAME_None)
73 {
74 return FText::FromName(ActorResourcePropertyName);
75 }
76 if (CostSource == ENodeCostSource::GameplayAttributeValue && ResourceGameplayAttribute.IsValid())
77 {
78 return FText::FromString(ResourceGameplayAttribute.GetName());
79 }
80 return LOCTEXT("UnknownResourceName", "Resource");
81 }
82};

84// Hash function for TMap support.
85inline uint32 GetTypeHash(const FNodeCostDefinition& Def)
86{
87 uint32 Hash = GetTypeHash(static_cast<uint8>(Def.CostSource));
88 if (Def.CostSource == ENodeCostSource::ActorIntegerProperty)
89 {
90 Hash = HashCombine(Hash, GetTypeHash(Def.ActorResourcePropertyName));
91 }
92 else if (Def.CostSource == ENodeCostSource::GameplayAttributeValue)
93 {
94 Hash = HashCombine(Hash, GetTypeHash(Def.ResourceGameplayAttribute));
95 }
96 return Hash;
97}
