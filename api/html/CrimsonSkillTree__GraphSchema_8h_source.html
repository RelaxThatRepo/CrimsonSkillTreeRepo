24 bool CheckForLoop(UEdGraphNode* StartNode, UEdGraphNode* EndNode)
25 {
26 VisitedNodes.Add(StartNode);
27 return TraverseNodes(EndNode);
28 }
29
30private:
37 bool TraverseNodes(UEdGraphNode* Node)
38 {
39 VisitedNodes.Add(Node);
40
41 for (auto MyPin : Node->Pins)
42 {
43 if (MyPin->Direction == EGPD_Output)
44 {
45 for (auto OtherPin : MyPin->LinkedTo)
46 {
47 UEdGraphNode* OtherNode = OtherPin->GetOwningNode();
48 if (VisitedNodes.Contains(OtherNode))
49 {
50 return false;
51 }
52 else if (!FinishedNodes.Contains(OtherNode))
53 {
54 if (!TraverseNodes(OtherNode))
55 return false;
56 }
57 }
58 }
59 }
60
61 VisitedNodes.Remove(Node);
62 FinishedNodes.Add(Node);
63 return true;
64 };
65
67 TSet<TObjectPtr<UEdGraphNode>> VisitedNodes;
69 TSet<TObjectPtr<UEdGraphNode>> FinishedNodes;
70};
